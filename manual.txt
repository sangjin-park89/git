cd.. #뒤로가기

git init # 폴더에 git 확인하기 or 넣어주기

git status # untracked files 출력되면 git에 추가해야한다.

git add (파일이름) # 파일이름은 탭 눌러서 추가할수도 있다.

git add -A # 폴더안에있는 모든 항목을 넣어준다.

git status 다시 명령해서 파랗게 뜨면 git이 관리하는 항목이 됐다.
하지만 아직 완벽히 관리하는건 아니다.

git commit -m "First Commit" # First Commit 가 위의 add 해준 파일의 주석이 된다.

git commit -m "Second Commit" # 다음 주석은 이렇게 설정해줄수있다.
